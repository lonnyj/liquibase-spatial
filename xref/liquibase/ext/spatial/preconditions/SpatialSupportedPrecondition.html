21  /**
22   * <code>SpatialSupportedPrecondition</code> checks the state of the database and determines if it
23   * has spatial support.
24   * 
25   * @author Lonny Jacobson
26   */
27  public class SpatialSupportedPrecondition implements Precondition {
28     @Override
29     public String getName() {
30        return "spatialSupported";
31     }
32  
33     @Override
34     public Warnings warn(final Database database) {
35        final Warnings warnings = new Warnings();
36        if (!(database instanceof DerbyDatabase || database instanceof H2Database
37              || database instanceof MySQLDatabase || database instanceof OracleDatabase || database instanceof PostgresDatabase)) {
38           warnings.addWarning(database.getDatabaseProductName()
39                 + " is not supported by this extension");
40        }
41        return warnings;
42     }
43  
44     @Override
45     public ValidationErrors validate(final Database database) {
46        final ValidationErrors errors = new ValidationErrors();
47        if (!(database instanceof DerbyDatabase || database instanceof H2Database
48              || database instanceof MySQLDatabase || database instanceof OracleDatabase || database instanceof PostgresDatabase)) {
49           errors.addError(database.getDatabaseProductName() + " is not supported by this extension");
50        }
51        return errors;
52     }
53  
54     @Override
55     public void check(final Database database, final DatabaseChangeLog changeLog,
56           final ChangeSet changeSet) throws PreconditionFailedException, PreconditionErrorException {
57        if (database instanceof DerbyDatabase || database instanceof H2Database) {
58           final TableExistsPrecondition precondition = new TableExistsPrecondition();
59           precondition.setTableName("geometry_columns");
60           precondition.check(database, changeLog, changeSet);
61        } else if (database instanceof PostgresDatabase) {
62           final ViewExistsPrecondition precondition = new ViewExistsPrecondition();
63           precondition.setSchemaName("public");
64           precondition.setViewName("geometry_columns");
65           precondition.check(database, changeLog, changeSet);
66        } else if (database instanceof OracleDatabase) {
67           final ViewExistsPrecondition precondition = new ViewExistsPrecondition();
68           precondition.setSchemaName("mdsys");
69           precondition.setViewName("user_sdo_geom_metadata");
70           precondition.check(database, changeLog, changeSet);
71        } else if (!(database instanceof MySQLDatabase)) {
72           final Throwable exception = new LiquibaseException(database.getDatabaseProductName()
73                 + " is not supported by this extension");
74           final ErrorPrecondition errorPrecondition = new ErrorPrecondition(exception, changeLog,
75                 this);
76           throw new PreconditionErrorException(errorPrecondition);
77        }
78     }
